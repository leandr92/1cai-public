```python
import tree_sitter
from tree_sitter import Language, Parser

class BSLParser:
    """–ü–∞—Ä—Å–µ—Ä BSL –∫–æ–¥–∞ –≤ Abstract Syntax Tree"""
    
    def __init__(self):
        # –ó–∞–≥—Ä—É–∂–∞–µ–º –≥—Ä–∞–º–º–∞—Ç–∏–∫—É BSL
        BSL_LANGUAGE = Language('build/bsl.so', 'bsl')
        self.parser = Parser()
        self.parser.set_language(BSL_LANGUAGE)
    
    def parse_file(self, code: str) -> dict:
        """
        –ü–∞—Ä—Å–∏—Ç BSL —Ñ–∞–π–ª –∏ –∏–∑–≤–ª–µ–∫–∞–µ—Ç —Å—Ç—Ä—É–∫—Ç—É—Ä—É
        
        Returns:
            {
                'functions': [...],
                'procedures': [...],
                'variables': [...],
                'queries': [...],
                'complexity': int
            }
        """
        tree = self.parser.parse(bytes(code, "utf8"))
        root_node = tree.root_node
        
        functions = []
        for node in root_node.children:
            if node.type == 'function_declaration':
                func_info = {
                    'name': self._get_function_name(node),
                    'parameters': self._get_parameters(node),
                    'return_type': self._get_return_type(node),
                    'body': self._get_function_body(node),
                    'start_line': node.start_point[0],
                    'end_line': node.end_point[0],
                    'complexity': self._calculate_complexity(node)
                }
                functions.append(func_info)
        
        return {
            'functions': functions,
            'total_complexity': sum(f['complexity'] for f in functions),
            'loc': len(code.split('\n'))
        }
```

```python
from github import Github
import hmac
import hashlib

class GitHubIntegration:
    """–ò–Ω—Ç–µ–≥—Ä–∞—Ü–∏—è —Å GitHub –¥–ª—è –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–æ–≥–æ review"""

        self.webhook_secret = webhook_secret

    def verify_webhook(self, payload: bytes, signature: str) -> bool:
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –ø–æ–¥–ø–∏—Å–∏ webhook –æ—Ç GitHub"""
        mac = hmac.new(
            self.webhook_secret.encode(),
            msg=payload,
            digestmod=hashlib.sha256
        )
        return hmac.compare_digest(
            'sha256=' + mac.hexdigest(),
            signature
        )
    
    async def handle_pull_request_event(self, event_data: dict):
        """–û–±—Ä–∞–±–æ—Ç–∫–∞ —Å–æ–±—ã—Ç–∏—è Pull Request"""
        
        action = event_data['action']
        
        # –†–µ–∞–≥–∏—Ä—É–µ–º —Ç–æ–ª—å–∫–æ –Ω–∞ opened –∏ synchronize (–Ω–æ–≤—ã–µ commits)
        if action not in ['opened', 'synchronize']:
            return
        
        pr_number = event_data['number']
        repo_name = event_data['repository']['full_name']
        
        # –ü–æ–ª—É—á–∞–µ–º PR
        repo = self.github.get_repo(repo_name)
        pr = repo.get_pull(pr_number)
        
        # –ü–æ–ª—É—á–∞–µ–º –∏–∑–º–µ–Ω–µ–Ω–Ω—ã–µ —Ñ–∞–π–ª—ã
        files = []
        for file in pr.get_files():
            if file.filename.endswith('.bsl'):
                # Download file content
                content = repo.get_contents(
                    file.filename,
                    ref=pr.head.sha
                ).decoded_content.decode('utf-8')
                
                files.append({
                    'filename': file.filename,
                    'content': content,
                    'diff': file.patch
                })
        
        # –í—ã–ø–æ–ª–Ω—è–µ–º AI review
        review_result = await self.reviewer.review_pull_request({
            'files_changed': files,
            'pr_number': pr_number,
            'author': pr.user.login,
            'description': pr.body
        })
        
        # –ü—É–±–ª–∏–∫—É–µ–º review –≤ GitHub
        await self._post_review_to_github(pr, review_result)
    
    async def _post_review_to_github(
        self,
        pr,
        review_result: dict
    ):
        """–ü—É–±–ª–∏–∫–∞—Ü–∏—è review –≤ GitHub"""

                'CHANGES_REQUESTED': 'REQUEST_CHANGES',
                'COMMENTED': 'COMMENT'

            comments=review_result['comments']
        )
        
        # –î–æ–±–∞–≤–ª—è–µ–º labels
        if review_result['metrics']['critical'] > 0:
            pr.add_to_labels('security-review-needed')
        
        if review_result['metrics']['high'] > 5:
            pr.add_to_labels('needs-refactoring')
```

```python
# src/api/middleware/tenant_context.py

from fastapi import Request, HTTPException
from starlette.middleware.base import BaseHTTPMiddleware
import jwt

class TenantContextMiddleware(BaseHTTPMiddleware):
    """
    Middleware –¥–ª—è —É—Å—Ç–∞–Ω–æ–≤–∫–∏ –∫–æ–Ω—Ç–µ–∫—Å—Ç–∞ tenant
    –ò–∑–≤–ª–µ–∫–∞–µ—Ç tenant_id –∏–∑ JWT —Ç–æ–∫–µ–Ω–∞ –∏ —É—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ—Ç –≤ –∫–æ–Ω—Ç–µ–∫—Å—Ç
    """
    
    async def dispatch(self, request: Request, call_next):
        # –ò–∑–≤–ª–µ–∫–∞–µ–º JWT –∏–∑ –∑–∞–≥–æ–ª–æ–≤–∫–∞
        auth_header = request.headers.get('Authorization')
        
        if not auth_header or not auth_header.startswith('Bearer '):
            raise HTTPException(status_code=401, detail="Missing authentication")
        
        token = auth_header.replace('Bearer ', '')
        
        try:
            # –î–µ–∫–æ–¥–∏—Ä—É–µ–º JWT
            payload = jwt.decode(
                token,
                settings.JWT_SECRET,
                algorithms=['HS256']
            )
            
            tenant_id = payload.get('tenant_id')
            user_id = payload.get('user_id')
            
            if not tenant_id:
                raise HTTPException(status_code=403, detail="Missing tenant_id in token")
            
            # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –≤ request state
            request.state.tenant_id = tenant_id
            request.state.user_id = user_id
            
            # –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º PostgreSQL session variable
            # –≠—Ç–æ –∞–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∏ –ø—Ä–∏–º–µ–Ω–∏—Ç RLS policies
            async with request.app.state.db_pool.acquire() as conn:
                await conn.execute(
                    f"SET app.current_tenant_id = '{tenant_id}'"
                )
            
            response = await call_next(request)
            return response
            
        except jwt.InvalidTokenError:
            raise HTTPException(status_code=401, detail="Invalid token")
```

```python
# src/api/usage_tracking.py

class UsageTracker:
    """–û—Ç—Å–ª–µ–∂–∏–≤–∞–Ω–∏–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è –¥–ª—è billing"""

        self,
        tenant_id: uuid.UUID,
        endpoint: str,
        tokens_used: int = 0
    ):
        """–ó–∞–ø–∏—Å—ã–≤–∞–µ–º –∫–∞–∂–¥—ã–π API call"""
        
        await self.redis.hincrby(
            f"usage:{tenant_id}:{datetime.now().strftime('%Y-%m')}",
            'api_calls',
            1
        )
        
        if tokens_used > 0:
            await self.redis.hincrby(
                f"usage:{tenant_id}:{datetime.now().strftime('%Y-%m')}",
                'tokens',
                tokens_used
            )
    
    async def check_usage_limits(
        self,
        tenant_id: uuid.UUID
    ) -> dict:
        """–ü—Ä–æ–≤–µ—Ä–∫–∞ –ª–∏–º–∏—Ç–æ–≤ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è"""
        
        # –ü–æ–ª—É—á–∞–µ–º –ø–ª–∞–Ω tenant
        tenant = await self.db.fetchrow(
            'SELECT plan, status FROM tenants WHERE id = $1',
            tenant_id
        )
        
        # –õ–∏–º–∏—Ç—ã –ø–æ –ø–ª–∞–Ω–∞–º
        limits = {
            'starter': {
                'api_calls': 10_000,
                'tokens': 1_000_000,
                'users': 5
            },
            'professional': {
                'api_calls': 50_000,
                'tokens': 5_000_000,
                'users': 20
            },
            'enterprise': {
                'api_calls': -1,  # Unlimited
                'tokens': -1,
                'users': -1
            }
        }
        
        plan_limits = limits[tenant['plan']]
        
        # –¢–µ–∫—É—â–µ–µ –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏–µ
        current_month = datetime.now().strftime('%Y-%m')
        usage_key = f"usage:{tenant_id}:{current_month}"
        
        current_usage = await self.redis.hgetall(usage_key)
        api_calls = int(current_usage.get('api_calls', 0))
        tokens = int(current_usage.get('tokens', 0))
        
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º –ª–∏–º–∏—Ç—ã
        if plan_limits['api_calls'] != -1 and api_calls >= plan_limits['api_calls']:
            return {
                'allowed': False,
                'reason': 'API calls limit exceeded',
                'current': api_calls,
                'limit': plan_limits['api_calls'],
                'upgrade_suggestion': 'professional' if tenant['plan'] == 'starter' else 'enterprise'
            }
        
        return {
            'allowed': True,
            'usage': {
                'api_calls': api_calls,
                'tokens': tokens
            },
            'limits': plan_limits,
            'usage_percent': {
                'api_calls': (api_calls / plan_limits['api_calls'] * 100) if plan_limits['api_calls'] != -1 else 0
            }
        }
```
